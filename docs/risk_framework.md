    # 估值条件（至少满足2项）
    "valuation_criteria": [
        "pe < industry_avg * 0.7",             # PE低于行业30%

## 5. 综合决策矩阵

### 5.1 CIO Agent 决策逻辑

```python
class InvestmentDecision:
    """投资决策"""
    
    action: str          # BUY / WATCH / SELL
    confidence: float    # 置信度 0-1
    reason: str          # 1句话原因
    
    quality_score: int   # 质量评分
    value_score: int     # 价值评分
    risk_score: int      # 风险评分（分数越高越危险）
    composite_score: int # 综合评分

DECISION_MATRIX = {
    # 风险优先原则
    "risk_first": True,
    
    # 决策规则
    "rules": [
        # 极高风险，直接卖出
        {"if": "risk_score >= 80", "then": "SELL", "reason": "极高风险"},
        
        # 高风险，谨慎
        {"if": "risk_score >= 60", "then": "WATCH", "reason": "高风险"},
        
        # 高质量 + 低估值 + 低风险 = 买入
        {"if": "quality_score >= 70 AND value_score >= 70 AND risk_score <= 30", "then": "BUY", "reason": "优质低估"},
        
        # 中等质量 + 低估值 + 低风险 = 观察
        {"if": "quality_score >= 50 AND value_score >= 60 AND risk_score <= 40", "then": "WATCH", "reason": "关注机会"},
        
        # 高估值 + 任何风险 = 卖出
        {"if": "value_score <= 30 AND risk_score >= 40", "then": "SELL", "reason": "高估且有风险"},
        
        # 默认：观察
        {"if": "default", "then": "WATCH", "reason": "条件不明确"}
    ]
}

# 综合评分公式
COMPOSITE_FORMULA = "quality_score * 0.3 + value_score * 0.3 - risk_score * 0.4"

# 综合评分区间
COMPOSITE_THRESHOLDS = {
    "strong_buy": 70,    # 综合评分 > 70 → 强烈买入
    "buy": 50,           # 综合评分 > 50 → 买入
    "watch": 20,         # 综合评分 > 20 → 观察
    "sell": 0            # 综合评分 <= 20 → 卖出
}
```
